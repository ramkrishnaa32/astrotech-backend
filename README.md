# astrotech-backend

# Run locally
npm install
npm start

# Get All Users
GET http://localhost:3000/api/users

# Get User by ID
GET http://localhost:3000/api/users/<user_id>

# Get All Astrologers
GET http://localhost:3000/api/astrologers

# Get Astrologer by ID
GET http://localhost:3000/api/astrologers/<astrologer_id>
